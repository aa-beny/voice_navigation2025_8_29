B) 沒麥克風 + 用 GPT（語意理解）
source ~/tb420250806/install/setup.bash
export OPENAI_API_KEY=你的金鑰
export AUDIO_FILES="/home/jetson/voice_tests/*.wav"
export USE_OPENAI=true
ros2 launch voice_nav2_assistant from_file.launch.py


提醒：你的 from_file.launch.py 目前固定 audio_mode='file'，所以用 ros2 launch 是跑讀檔模式；
要用麥克風，請用 ros2 run（下面有指令）。

C) 有麥克風 + 不用/用 GPT（用 ros2 run 才吃得到鍵盤 Enter）
# 不用 GPT
PL=$(ros2 pkg prefix voice_nav2_assistant)/share/voice_nav2_assistant/config/places.yaml
ros2 run voice_nav2_assistant voice_nav_node --ros-args \
  -p places_file:=$PL -p audio_mode:=mic -p use_openai:=false

# 用 GPT
export OPENAI_API_KEY=你的金鑰
ros2 run voice_nav2_assistant voice_nav_node --ros-args \
  -p places_file:=$PL -p audio_mode:=mic -p use_openai:=true
# 視窗顯示：🎤 麥克風模式：按 Enter 錄音 3 秒，q 離開


（可選）調整錄音秒數 / 取樣率：

export MIC_DURATION=3    # 秒
export MIC_RATE=16000    # Hz
